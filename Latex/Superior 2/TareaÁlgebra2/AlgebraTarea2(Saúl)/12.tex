\section{Sean $a,b \in \mathbb{Z}$ son tales que toda solución de $ax+by = 35$ es de la forma } %Cierre de titulo de sección
    \begin{align}
        \notag a7 + b7 &= 35 \\
        \notag \frac{b}{(a;b)} = -3 &\rightarrow 6 = -3d \\
        \notag \frac{a}{(a;b)} = 4 &\rightarrow a = 4d \\
        \notag 28d - 21d &= 35  \\
        \notag 7d &= 35 \\
        \notag \therefore d = 5
    \end{align}
    $\therefore b = -15 \wedge a = 20 \qquad \blacksquare$
